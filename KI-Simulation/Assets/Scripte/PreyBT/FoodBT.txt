tree("Root")
	repeat
		mute
			fallback
				tree "flee"
				tree "eating"
				tree "wander"
				
tree("wander")
	sequence		
		wander
		moveToDestination
	
tree("eating")
	while isHungry
		sequence
			wander
			searchFood
			moveToDestination
			while hungerCheck()
				sequence
					eat(10)
					Wait 2.0

tree("flee")
	while inDanger	
		sequence
			flee
			moveToDestination
